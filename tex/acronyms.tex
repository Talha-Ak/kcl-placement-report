% Command to create a glossary entry with correspondent acronym.
% Args : 1: label, 2: short name, 3: long name, 4: description
\newcommand{\newglossaryentrywithacronym}[4]{
    % The glossary entry the acronym links to
    \newglossaryentry{#1_gls}{
        name={#3},
        description={#4}
    }

    % Acronym pointing to glossary
    \newglossaryentry{#1}{
        type=\acronymtype,
        name={#2},
        description={#3},
        first={#3 (#2)\glsadd{#1_gls}},
        see=[Glossary:]{#1_gls}
    }
}
